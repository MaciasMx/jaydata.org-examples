YQL Feed

var dataContext = new $data.Yahoo.YQLContext({ name: 'YQL' });
dataContext.onReady(function (context) {
    context.Feed
    .filter(function (h) { return h.url == 'http://stackoverflow.com/feeds' })
    .tojQueryTemplate('#feedTemplate', '#result');
});
