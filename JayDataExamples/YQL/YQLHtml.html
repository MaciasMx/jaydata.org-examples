How to get the top news from BBC?


var dataContext = new $data.Yahoo.YQLContext({ name: 'YQL' });
dataContext.onReady(function (context) {
    context.Html
        .filter(function (h) { return h.url == 'http://www.bbc.com/' && h.xpath == '//div[@id="carousel_items"]/dl' })
        .forEach();
});
