Advanced RSS reader

Define new context with extended element type

$data.EntityContext.extend("MyYQLContext", {
    Rss: {
        tableName: 'rss',
        dataType: $data.EntitySet,
        elementType: $data.Yahoo.types.YQLRss.extend("MyYQLRss", {
            category: { type: 'array' }
        })
    }
});

Use this context for extended properties

var myRssContext = new MyYQLContext({ name: 'YQL' });
myRssContext.onReady(function (context) {
    context.Rss
        .filter(function (r) { return r.url == 'http://coder.io/tag/javascript.rss' && r.title.contains('javascript'); })
        .tojQueryTemplate('#rssTemplate', '#result');
});
